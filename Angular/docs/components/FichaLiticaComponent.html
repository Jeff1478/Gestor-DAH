  OnSubmit(){

    this._liticoService.create(this.litico).subscribe(
      
      response => {
       if(response.status == 'success'){
         this.status = 'success';
         this.litico = response.litico;
      
 
           Swal.fire('Artefacto Creado', 'Guardado Correctamente!', 'success')
       
        
         this._router.navigate(['/pag-lit']);
       }else{
         this.status = 'error'
       }
      },
      error => {
       console.log(error);
       this.status = 'error';
      }
    );
